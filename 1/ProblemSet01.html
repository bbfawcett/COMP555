def loadFasta(filename):
    """ Parses a classically formatted and possibly 
        compressed FASTA file into two lists. One of 
        headers and a second list of sequences.
        The ith index of each list correspond."""
    if (filename.endswith(".gz")):
        fp = gzip.open(filename, 'r')
    else:
        fp = open(filename, 'r')
    # split at headers
    data = fp.read().split('>')
    fp.close()
    # ignore whatever appears before the 1st header
    data.pop(0)     
    headers = []
    sequences = []
    for sequence in data:
        lines = sequence.split('\n')
        headers.append(lines.pop(0))
        # add an extra "+" to make string "1-referenced"
        sequences.append('+' + ''.join(lines))
    return (headers, sequences)

def shortestMissingKmer(seq):
    """ Returns the value of k for the shortest missing 
         k-mer, and a list of all such k-mers"""
    k, missing = 0, 0
    original = seq
    bases = ['A', 'T', 'C', 'G']
    
    allKmers = True
    kmers = bases
    k = 1
    
    
    while (allKmers):
        seq = original + original[1:k-1]
        kmerDict = {}
        for i in range(1,len(seq)-k+1):
            kmer = seq[i:i+k]
            kmerDict[kmer] = kmerDict.get(kmer,0) + 1
            
            
        if (4**k - len(kmerDict) != 0):
            missing = 4**k - len(kmerDict)
            allKmers = False
        else:
            k += 1
            
        
    
    return k, missing

def nonrepeatingKmers(seq):
    unique90 = 0
    notFound = True
    i = 1
    
    while notFound:
        output = kmerFrequency(seq,i)
        denom = 0
        for x in range(1,len(output)-1):
            denom += x*output[x]
        
        if ((output[1]/denom) > 0.9):
            unique90 = i
            notFound = False
        else:
            i += 1
        
    return unique90

def getGCdiff(sequence, start):
    halflen = len(sequence)//2
    terC = start + halflen
    # handle genome's circular nature
    if (terC > len(sequence)):
        terC = terC - len(sequence) + 1
    if (start <= 0):
        start = start + len(sequence) - 1
    if (start > len(sequence)):
        start = start - len(sequence) + 1
    if (terC <= 0):
        terC = terC + len(sequence) - 1
    if (start == 0):
        start = 1
    if (terC > start):                                   # case 1: ----S========T---->
        G = 2*sequence[start:terC].count('G') - sequence.count('G')
        C = 2*sequence[start:terC].count('C') - sequence.count('C')
    else:                                                # case 2: ====T--------S====>
        G = sequence.count('G') - 2*sequence[terC:start].count('G')
        C = sequence.count('C') - 2*sequence[terC:start].count('C')
    return start, (G - C)

def GCSkew(genome):
    x = []
    y = []
    for i in range(1,len(genome),500):
        addx, addy = getGCdiff(genome, i)
        x.append(addx)
        y.append(addy)
    return x, y



def findOriR(genome, gcMax):
    xSmall = []
    ySmall = []
    for i in range(gcMax-750, gcMax+750, 1):
        addx, addy = getGCdiff(genome, i)
        xSmall.append(addx)
        ySmall.append(addy)
        
        
    xLow = xSmall[ySmall.index(max(ySmall))] - 500
    xHigh = xSmall[ySmall.index(max(ySmall))] + 500
    
    if (xLow <= 0):
        xLow = xLow + len(genome) - 1
    if (xLow >= len(genome)-1):
        xLow = xLow - (len(genome)-1)
        
    if (xHigh <= 0):
        xHigh = xHigh + len(genome) - 1
    if (xHigh >= len(genome)-1):
        xHigh = xHigh - (len(genome)-1)

    return xSmall, ySmall, xLow, xHigh
